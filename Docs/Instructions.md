    r1

    r4 - и не занимают место в памяти
    r5 - они могут хранить в себе очень большие числа
    r6 - и с плавающей точкой.
    r7 - но ограничение по размеру числа тоже есть.

## инструкции !


# out - инструкция для вывода текста в консоль
    out "welcome to hell" - вывести текст
    out r1 - вывести значение регистра
    out var_name - вывести значение переменной

# next - инструкция для вывода, переход на следующую строку
    out "welcome to hell"
    next
    out "vlad"

    результат:

    welcome to hell
    vlad

# создание переменных
    db name 0    - создать переменную 1байт с значением 0
    dw name 0    - создать переменную 2байт с значением 0
    dd name 0    - создать переменную 4байт с значением 0 + плавающая точка
    dq name 0    - создать переменную 8байт с значением 0 + плавающая точка
    ds name ""    - создать переменную string с значением ""

    важно запомнить!
    строка весит столько байт, сколько в ней символов
    ds name "welcome to hell" - 15 байт т.к. 15 символов.

    и переменной нельзя присвоить значение другой переменной или регистра, только после обьявления

    пример
    mov name r1 - заменяем значение name на значение регистра
    mov name name - заменяем значение name на значение переменной name


# clear - очистка памяти
    clear registres - очищает все регистры процессора
    clear screen - очищает экран
    clear r1 - очищает регистр р1
    clear name - очищает ячейку памяти занятую переменной name (ее больше нет)

    ОЧЕНЬ ВАЖНО НЕ ЗАБЫВАТЬ ОЧИЩАТЬ ПАМЯТЬ, ХОТЯ БЫ В stop:
    ИНАЧЕ БУДЕТ RAM IS FULL 


# mov - инструкция для перемещения
    mov r1 0
    mov r1 r1
    mov r1 name

    mov name 0
    mov name r1
    mov name name
    mov name "text"

    заменяет значение первого аргумента, на значение второго

# inp - ввод пользователя
    
    inp r1 - записывает ввод в регистр р1
    inp name - записывает ввод в переменную name
    

# add - инструкция для прибавления
    add r1 1
    add r1 r1
    add r1 var_name
    add var_name 1
    add var_name "hello"
    add var_name r1
    add var_name var_name - ВАЖНО ЗАПОМНИТЬ! СКЛАДЫВАТЬ МОЖНО ДВЕ ПЕРЕМЕННЫХ, ЕСЛИ ОНИ ОДНОГО ТИПА (пример db и db)

# sub - инструкция для убавления
    тоже самое, что и add, но нельзя вычесть что либо из переменной стринг

# mul - инструкция для умножения
    тоже самое, что и add, но нельзя умножить строку

# div - инструкция для деления
    тоже самое, что а add, но нельзя делить строку

# call - инструкция для вызова блока
    call block_name - вызывает блок, а потом возвращается обратно после выполнения
    
    пример:

    .block name1:
        call name2
        out "!" - когда блок от call закончится, то вернется сюда и выведет плюсом !
    end

    .block name2:
        out "welcome to hell"
        ret - ВЕРНУТЬСЯ, ОЧЕНЬ ВАЖНАЯ ИНСТРУКЦИЯ ПРИ ИСПОЛЬЗОВАНИИ call И НЕ ИСПОЛЬЗОВАТЬ ПРИ ВЫЗОВЕ ЧЕРЕЗ go
    end

# go - инструкция для перехода к блоку
    go block_name - переходит к блоку не возвращаясь больше.

    пример:

    .block name1:
        go name2
        out "!" - это не выполнится.
    end

    .block name2:
        out "welcome to hell"
    end

    результат

    welcome to hell

# cmp - сравнить два аргумента

    cmp r1 0 - сраниваваем регистр р1 и 0
    cmp r1 r1 - сравниваем регистр р1 и р1
    cmp r1 name - сравниваем регистр р1 и нейм

    cmp name 0 - сравниванием переменную нейм и 0
    cmp name "" - сравниваем переменную нейм и ""
    cmp name r1 - сравниваем переменную нейм и регистр р1
    cmp name name - сравниваем переменные нейм и нейм

    результат сохраняется в стэке, и проверить его можно с помощью условий
        ife
        ifn
        ifl
        ifg

    примечание: если вы сравниваете строковые переменные, то сравнивается их ДЛИННА!

# ife - если результат сравнения показал равеноство - аргумент1 == аргумент2

    ife go name_block - если true, то мы go на блок
    ife call name_block - так же, но call


# ifn - если результат сравнения показал неравеноство - аргумент1 != аргумент2

    ifn go name_block - если true, то мы go на блок
    ifn call name_block - так же, но call


# ifg - если результат сравнения показал больше - аргумент1 > аргумент2

    ifg go name_block - если true, то мы go на блок
    ifg call name_block - так же, но call


# ifl - если результат сравнения показал меньше - аргумент1 < аргумент2

    ifl go name_block - если true, то мы go на блок
    ifl call name_block - так же, но call
