r1 - double

# Предназначения регистров

------------------------------------------------------------------------------------------------------------------
# rnd
rnd - регистр, который может хранить в себе число double, но его первичная задача хранить в себе
число заданное рандомом.

пример:

randd 0, 1000 

out, rnd

вывод: рандомное число от 0 до 1000, например 50

------------------------------------------------------------------------------------------------------------------



------------------------------------------------------------------------------------------------------------------
# rnr
rnr - регистр, который может хранить в себе число double, но его первичная задача хранить в себе
остаток от деления

пример:

mov r1, 50
mov r2, 20

div r1, r2

out, rnr

вывод: 10, т.к. остаток от деления 50 на 20 будет 10.

-------------------------------------------------------------------------------------------------------------------



------------------------------------------------------------------------------------------------------------------
# rvc
rvc - регистр, который может хранить в себе число double, но его первичная задача хранить в себе
дистанцию между двумя векторами.

пример:

distance obj1, obj2     - высчитываем дистанцию между двумя векторами

результат сохраняется в регистр rvc
